Jquery.i18now

This plugin helps to customize easily client-side date and time formatting.

Download as .zip Download as .tar.gz View on GitHub

Demo:

See here: http://mbitto.com/projects.html

Usage:

Get translations and formats from JSON file write them directly to a JavaScript file:

{
    "usa" : {
        "D" : ["Sun", "Mon", "Tue", "Wen", "Thu", "Fri", "Sat"],
        "l" : ["Sunday", "Monday", "Tuesday", "Wednesday", "Thursday", "Friday", "Saturday"],
        "M" : ["Jan", "Feb", "Mar", "Apr", "May", "Jun", "Jul", "Aug", "Sep", "Oct", "Nov", "Dec"],
        "F" : ["January", "February", "March", "April", "May", "June", "July",
            "August", "September", "October", "November", "December"],
        "format" : "%l, %F %d, %Y %h:%i:%s %A"
    },
    "deu" : {
        "D" : ["So.", "Mo.", "Di.", "Mi.", "Do.", "Fr.", "Sa."],
        "l" : ["Sonntag", "Montag", "Dienstag", "Mittwoch", "Donnerstag", "Freitag", "Samstag"],
        "M" : ["Jan", "Feb", "Mär", "Apr", "Mai", "Jun", "Jul", "Aug", "Sep", "Okt", "Nov", "Dez"],
        "F" : ["Januar", "Februar", "März", "April", "Mai", "Juni", "Juli",
            "August", "September", "Oktober", "November", "Dezember"],
        "format" : "%l, %d. %F %Y %H:%i:%s"
    },
    "jpn" : {
        "D" : ["日", "月", "火", "水", "木", "金", "土"],
        "l" : ["日曜日", "月曜日", "火曜日", "水曜日", "木曜日", "金曜日", "土曜日"],
        "M" : ["1月", "2月", "3月", "4月", "5月", "6月", "7月", "8月", "9月", "10月", "11月", "12月"],
        "F" : ["1月", "2月", "3月", "4月", "5月", "6月", "7月", "8月", "9月", "10月", "11月", "12月"],
        "format" : "%Y年%F%d日%l%G時%i分%s秒"
    }
}

Initialize i18Now:

$("#date-usa").i18Now(formats['usa']);

Update it every n seconds

$("#date-usa").i18Now('update', 1);

Create a custom date and time format with default translation (english)

$("#date-custom").i18Now({
    format : "It's %l, %j/%F/%y"
});

Show any date and time that you like, using JavaScript Date object

$("#date-custom").i18Now('setCustomDate', new Date(1986, 2, 24));
